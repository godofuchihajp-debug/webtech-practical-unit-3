<script>
let num = 5;

// if-else
if (num % 2 == 0) {
    document.write("Even Number<br>");
} else {
    document.write("Odd Number<br>");
}

// for loop
for (let i = 1; i <= 3; i++) {
    document.write("For Loop: " + i + "<br>");
}

// while loop
let j = 1;
while (j <= 3) {
    document.write("While Loop: " + j + "<br>");
    j++;
}

// do-while loop
let k = 1;
do {
    document.write("Do-While Loop: " + k + "<br>");
    k++;
} while (k <= 3);
</script>
