<script>
function showScope() {
    var a = 10;     // function scope
    let b = 20;     // block scope
    const c = 30;   // constant value

    document.write("var a = " + a + "<br>");
    document.write("let b = " + b + "<br>");
    document.write("const c = " + c + "<br>");
}

showScope();
</script>
